CREATE TABLE IF NOT EXISTS `users` (
  `id` int(30) UNSIGNED NOT NULL AUTO_INCREMENT,
  `userid` varchar(20) NOT NULL,
  `membership` tinyint(1) NOT NULL DEFAULT '0',
  `RealIp` varchar(25)  NOT NULL,
  `browser` varchar(60)  NOT NULL,
  `is_admin` tinyint(1) NOT NULL DEFAULT '0',
  `banned_user` tinyint(1) NOT NULL DEFAULT '0',
   `buy_time` varchar(35) NOT NULL,
  `Introduced` varchar(20)  NOT NULL,
  `request_time` varchar(35) NOT NULL,
  `time_created` varchar(35) NOT NULL,
  `time_updated` varchar(35) NOT NULL,
  `payments` varchar(35)  NOT NULL DEFAULT '400000',
  `accept_time` varchar(35) NOT NULL,
  PRIMARY KEY (`id`)
